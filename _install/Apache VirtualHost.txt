<VirtualHost *:80>
	ServerAdmin john@defvayne23.com
	ServerName {DOMAIN}
	ServerAlias {SUB}.{DOMAIN}

	DocumentRoot /var/hosts/{DOMAIN}/{SUB}/public_html/
	<Directory />
		Options FollowSymLinks
		AllowOverride None
	</Directory>
	<Directory /var/hosts/{DOMAIN}/{SUB}/public_html/>
		Options Indexes FollowSymLinks MultiViews
		AllowOverride All
		Order allow,deny
		allow from all

		RewriteEngine On
		RewriteBase /

		RewriteRule ^editor(.*)$ editor$1 [PT]
		RewriteRule ^img(.*)$ img$1 [PT]
		RewriteRule ^css(.*)$ css$1 [PT]
		RewriteRule ^scripts(.*)$ scripts$1 [PT]

		RewriteCond %{HTTP_HOST} !^www.(.*) [NC]
		RewriteCond %{HTTP_HOST} !^[a-z0-9]+\.[a-z0-9]+\.[a-z0-9]+ [NC]
		RewriteRule ^(.*)$ http:\/\/www.%{HTTP_HOST}/$1 [R=301,L]

		#RewriteCond %{REQUEST_URI} !\.(xml|js|ico|txt|gif|jpg|png|css|pdf|zip|fla|swf|htc)$ [NC]
		RewriteRule ^.*$ index.php
	</Directory>

	LogLevel warn
	ErrorLog  /var/hosts/{DOMAIN}/{SUB}/.logs/error.log
	CustomLog /var/hosts/{DOMAIN}/{SUB}/.logs/access.log combined
</VirtualHost>
